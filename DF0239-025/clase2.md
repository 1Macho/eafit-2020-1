$$
\vec F _ {i0} = {{Kq_iq_0(\vec r - \vec r_i)} \over {|\vec r - \vec r_i|^3}}
$$

$$
\vec F = \sum ^ {n} _ {i=1} {{Kq_iq_0(\vec r - \vec r_i)} \over {|\vec r - \vec r_i|^3}}
$$
